using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Observado : MonoBehaviour
{
    public static Observado Instance;

    public List<OnNotified> ObserversAmount;
    void Awake()
    {
         
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return;
        }

        Instance = this;
        ObserversAmount = new List<OnNotified>();
    }

}
